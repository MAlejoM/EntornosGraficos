<?php
$a = array( 'color' => 'rojo',
            'sabor' => 'dulce',
            'forma' => 'redonda',
            'nombre' => 'manzana',
            4
        );
?>
------------------------------------------------------------------------------------------------------------
<?php
$a['color'] = 'rojo';
$a['sabor'] = 'dulce';
$a['forma'] = 'redonda';
$a['nombre'] = 'manzana';
$a[] = 4;
?>
------------------------------------------------------------------------------------------------------------
Estos dos codigos son equivalentes, ya que son dos formas distintas de crear un arreglo.
